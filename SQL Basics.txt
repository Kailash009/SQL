>> INSERT,UPDATE,DELETE
3. DCL. (Data Control Language)
>> GRANT, REVOKE.
4. TCL (Transaction Control Language)
>> BEGIN TRANSACTION,COMMIT,ROLLBACK,SAVEPOINT.
5. DQL (Data Query Language)
>> SELECT.

SELECT * from tbl_Employee where salary>30000

-- WAQ TO GET ALL THE EMPLOYEE WHOSE SALARY IS not 45000.

select * from tbl_Employee where salary <> 45000 

-- WAQ TO GET ALL THE EMPLOYEE WHOSE SALARY IS 36000 AND NAME Prateek
SELECT * FROM tbl_Employee where salary=36000 and ename='Prateek' or ename='ashish'

select * from tbl_Employee where ename='ashish' or eaddress='patel nagar' or age=10

-- WAQ TO GET ALL THE EMPLOYEE WHOSE SALARY IS not 45000.
select * from tbl_Employee where not salary=45000

-- WAQ TO GET ALL THE EMPLOYEE DETAILS WHOSE NAME IS ASHISH,PRATEEK,RATAN
select * from tbl_Employee where ename in('ashish','prateek','ratan')
select * from tbl_Employee where ename not in('ashish','prateek','ratan')

-- WAQ TO GET ALL THE EMPLOYEE WHOSE SALARY IS BETWEEN 20000 TO 40000.
SELECT * FROM tbl_Employee where salary between 20000 and 40000

-- WAQ TO GET ALL THE EMPLOYEE WHOSE SALARY IS not BETWEEN 20000 TO 40000.
SELECT * FROM tbl_Employee where salary not between 20000 and 40000


-- waq to get top 3 record from the employee.
select top 3 * from tbl_Employee

-- waq to get ALL THE EMPLOYEE IN ASCENDING ORDER.
select * from tbl_Employee order by salary desc

-- waq to get top 3 latest record from the employee.
select top 3 * from tbl_Employee order by eid desc


select * from tbl_Employee where ename like 'S_b_a_'
-- waq to get all the employees whose name start from a.
select * from tbl_Employee where ename like 'a%'

-- waq to get all the employees whose name ends from i.
select * from tbl_Employee where ename like '%a'
